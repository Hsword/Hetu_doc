[docs]    def compute(self, input_vals, output_val, stream_handle=None):
        if self.on_cpu:
            if DNNL_LIB['DnnlSigmoid']:
                cpu_sigmoid(input_vals[0], output_val)
            else:
                output_val[:] = 1.0/(1.0+1.0/np.exp(input_vals[0].asnumpy()))
        else:
            sigmoid(input_vals[0], output_val, stream_handle)



[docs]    def gradient(self, output_grad):
        #ds=s(1-s)
        grad_A=sigmoid_op(self.inputs[0], ctx=output_grad.ctx)*(1+ -1*sigmoid_op(self.inputs[0], ctx=output_grad.ctx))
        return [grad_A*output_grad]
